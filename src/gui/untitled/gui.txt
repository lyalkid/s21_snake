    Q_UNUSED(e)
    QPainter p(this);
    p.drawRect(0, 0, 300, 600);

    GameInfo_t inf = updateCurrentState();

void MainWindow::keyPressEvent(QKeyEvent *e) {
    UserAction_t action;
    switch (e->key()) {
    case Qt::Key_Up:
        action = Up;
        break;
    case Qt::Key_Down:
        action = Down;
        break;
    case Qt::Key_Left:
        action = Left;
        break;
    case Qt::Key_Right:
        action = Right;
        break;
    case 'p':
        action = Pause;
        break;
    case 'n':
        action = Start;
        break;
    case 'q':
        action = Terminate;

        break;
    case 'f':
        action = Action;
        break;
    default:
        // action = (UserAction_t)NONE_ACTION;
        break;

    }
    get_data()->current_action = action;
     userInput(action, false);
    update_draw();
     if(action == Terminate){
        QWidget::close();
    }
}
